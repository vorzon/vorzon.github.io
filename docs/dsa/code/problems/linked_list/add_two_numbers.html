2. 两数相加open in new window

# 题解

/**
 * Definition for singly-linked list.
 * type ListNode struct {
 *     Val int
 *     Next *ListNode
 * }
 */
func addTwoNumbers(l1 *ListNode, l2 *ListNode) *ListNode {
    var head, tail *ListNode
    temp := 0

    for l1 != nil || l2 != nil {
        v1, v2 := 0, 0
        if l1 != nil {
            v1 = l1.Val
            l1 = l1.Next
        }

        if l2 != nil {
            v2 = l2.Val
            l2 = l2.Next
        }

        sum := v1 + v2 + temp
        temp = sum / 10
        sum = sum % 10

        if head == nil {
            head = &ListNode{
                Val: sum,
            }
            tail = head
        } else {
            tail.Next = &ListNode{
                Val: sum,
            }
            tail = tail.Next
        }

    }

    if temp > 0 {
        tail.Next = &ListNode{Val: temp,}
    }

    return head
}
